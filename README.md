# Nodejs-helloworld

# Install Git:
    yum install git -y
# Install npm:
    sudo yum install -y gcc-c++ make
    curl -sL https://rpm.nodesource.com/setup_13.x | sudo -E bash -
    sudo yum install -y nodejs
# Build Nodejs:
    npm install
# Run Nodejs application using below command:
    node server.js
# Check Nodejs Application in UI:
  ip-address:8080
